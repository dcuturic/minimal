API Documentation

        
The API Request Builder functionality can also be accessed via API.

        
        

            Endpoint
            POST /api/minimal-solutions/api_request_builder
        

        
        

            Request Example
            
{
  "method": "POST",
  "url": "https://api.example.com/v1/users",
  "headers": {
    "Authorization": "Bearer token",
    "Content-Type": "application/json"
  },
  "body": {
    "name": "John Doe",
    "email": "john@example.com"
  }
}

        

        
        

            Response Example
            
{
  "status": "success",
  "data": {
    "preview": "POST /v1/users HTTP/1.1\nHost: api.example.com\nAuthorization: Bearer token\nContent-Type: application/json\n\n{\n  \"name\": \"John Doe\",\n  \"email\": \"john@example.com\"\n}",
    "method": "POST",
    "url": "https://api.example.com/v1/users",
    "headers": {
      "Authorization": "Bearer token",
      "Content-Type": "application/json"
    },
    "body": {
      "name": "John Doe",
      "email": "john@example.com"
    }
  }
}
